  <script>
    // Email pieces stored in JS (so not visible in plain HTML source)
    (function(){
      const user = "fatimalorraine23";
      const domain = "gmail.com";
      const subject = encodeURIComponent("Job Opportunity / Inquiry");
      const bodyDefault = encodeURIComponent("Hello Fatima,\n\nI would like to discuss a job opportunity with you. Please let me know a convenient time to connect.\n\nBest regards,\n");

      function composeMail(){
        const mail = user + "@" + domain;
        window.location.href = "mailto:" + mail + "?subject=" + subject + "&body=" + bodyDefault;
      }

      const btn = document.getElementById('contactBtn');
      btn.addEventListener('click', function(e){
        e.preventDefault();
        // small confirmation to avoid accidental clicks
        if(confirm("Open email draft to contact Fatima?")) composeMail();
      });

      // allow replacing placeholder image by drag & drop / file select (optional convenience)
      const photo = document.getElementById('profilePhoto');
      photo.style.cursor = 'pointer';
      photo.title = "Click to replace photo";
      photo.addEventListener('click', function(){
        // create file input on the fly
        const input = document.createElement('input');
        input.type = 'file';
        input.accept = 'image/*';
        input.onchange = function(e){
          const f = e.target.files[0];
          if(!f) return;
          const reader = new FileReader();
          reader.onload = function(ev){ photo.src = ev.target.result; };
          reader.readAsDataURL(f);
        };
        input.click();
      });
    })();
  </script>
